## Subscribe and Bind

::code-group

```js [JavaScript]
const channel = client.subscribe("public-updates");
channel.bind("price-updated", (data) => {
  console.log(data);
});
```

```swift [Swift]
let channel = client.subscribe("public-updates")
channel.bind("price-updated") { data, _ in
    print(data ?? "")
}
```

```kotlin [Kotlin]
val channel = client.subscribe("public-updates")
channel.bind("price-updated") { data, _ ->
    println(data)
}
```

```dart [Flutter]
final channel = client.subscribe('public-updates');
channel.bind('price-updated', (data, _) {
  print(data);
});
```

## Connect Lifecycle

::code-group

```js [JavaScript]
client.connection.bind("connected", () => {
  console.log("connected", client.connection.socket_id);
});
client.connect();
```

```swift [Swift]
client.connection.bind("connected") { data in
    print("connected", data ?? "")
}
client.connect()
```

```kotlin [Kotlin]
client.connection.bind("connected") { data ->
    println("connected $data")
}
client.connect()
```

```dart [Flutter]
client.connection.bind('connected', (data) {
  print('connected $data');
});
client.connect();
```

## Private/Presence Auth

::code-group

```js [JavaScript]
const client = new SockudoClient("app-key", {
  wsHost: "127.0.0.1",
  wsPort: 6001,
  forceTLS: false,
  channelAuthorization: {
    endpoint: "https://api.example.com/sockudo/auth",
    transport: "fetch",
  },
});
```

```swift [Swift]
let client = try SockudoClient(
    "app-key",
    options: .init(
        cluster: "local",
        forceTLS: false,
        wsHost: "127.0.0.1",
        wsPort: 6001,
        channelAuthorization: .init(
            endpoint: "https://api.example.com/sockudo/auth"
        )
    )
)
```

```kotlin [Kotlin]
val client = SockudoClient(
    "app-key",
    SockudoOptions(
        cluster = "local",
        forceTls = false,
        wsHost = "127.0.0.1",
        wsPort = 6001,
        channelAuthorization = ChannelAuthorizationOptions(
            endpoint = "https://api.example.com/sockudo/auth",
        ),
    ),
)
```

```dart [Flutter]
final client = SockudoClient(
  'app-key',
  SockudoOptions(
    cluster: 'local',
    forceTls: false,
    wsHost: '127.0.0.1',
    wsPort: 6001,
    channelAuthorization: const ChannelAuthorizationOptions(
      endpoint: 'https://api.example.com/sockudo/auth',
    ),
  ),
);
```

## Filtered Subscriptions (V2)

::code-group

```js [JavaScript]
import { Filter } from "@sockudo/client/filter";

const channel = client.subscribe("market-data", {
  filter: Filter.and(
    Filter.eq("event_type", "trade"),
    Filter.gte("price", "100")
  ),
});
```

```swift [Swift]
let channel = client.subscribe(
    "market-data",
    options: .init(
        filter: .and([
            .eq("event_type", "trade"),
            .gte("price", "100")
        ])
    )
)
```

```kotlin [Kotlin]
val channel = client.subscribe(
    "market-data",
    SubscriptionOptions(
        filter = Filter.and(
            Filter.eq("event_type", "trade"),
            Filter.gte("price", "100"),
        ),
    ),
)
```

```dart [Flutter]
final channel = client.subscribe(
  'market-data',
  options: SubscriptionOptions(
    filter: Filter.and(<FilterNode>[
      Filter.eq('event_type', 'trade'),
      Filter.gte('price', '100'),
    ]),
  ),
);
```

## Delta Compression (V2)

::code-group

```js [JavaScript]
const channel = client.subscribe("ticker:btc", {
  delta: { enabled: true, algorithm: "xdelta3" },
});
```

```swift [Swift]
let channel = client.subscribe(
    "ticker:btc",
    options: .init(
        delta: .init(enabled: true, algorithm: .xdelta3)
    )
)
```

```kotlin [Kotlin]
val channel = client.subscribe(
    "ticker:btc",
    SubscriptionOptions(
        delta = ChannelDeltaSettings(enabled = true, algorithm = DeltaAlgorithm.xdelta3),
    ),
)
```

```dart [Flutter]
final channel = client.subscribe(
  'ticker:btc',
  options: const SubscriptionOptions(
    delta: ChannelDeltaSettings(
      enabled: true,
      algorithm: DeltaAlgorithm.xdelta3,
    ),
  ),
);
```

## Client Events

Client event names must start with `client-`. Only works on private channels.

::code-group

```js [JavaScript]
const channel = client.subscribe("private-chat");
channel.trigger("client-message", { text: "hello" });
```

```swift [Swift]
let channel = client.subscribe("private-chat")
try channel.trigger("client-message", data: ["text": "hello"])
```

```kotlin [Kotlin]
val channel = client.subscribe("private-chat")
channel.trigger("client-message", mapOf("text" to "hello"))
```

```dart [Flutter]
final channel = client.subscribe('private-chat');
channel.trigger('client-message', <String, Object>{'text': 'hello'});
```

## Global Events

::code-group


```swift [Swift]
}
```

```kotlin [Kotlin]
client.bindGlobal { eventName, data ->

}
```

```dart [Flutter]
client.bindGlobal((eventName, data) {

});
```
